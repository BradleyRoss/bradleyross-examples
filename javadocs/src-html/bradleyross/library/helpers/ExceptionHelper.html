012public class ExceptionHelper {
013        public static final int FATAL = 1;
014        public static final int ERROR = 2;

017        public static final int DEBUG = 5;
018        public static final int TRACE = 6;
019        protected org.apache.log4j.Logger apacheLogger = null;
020        protected org.slf4j.Logger slf4jLogger = null;
021        public ExceptionHelper(org.apache.log4j.Logger logger) {
022                apacheLogger = logger;
023        }
024        public ExceptionHelper(org.slf4j.Logger logger) {
025                slf4jLogger = logger;
026        }
027        public void logException(int level, List<String> strings, Exception e) {
028                StringWriter writer = new StringWriter();
029                PrintWriter out = new PrintWriter(writer);
030                Iterator<String> iter = strings.iterator();
031                while (iter.hasNext()) {
032                        out.println(iter.next());
033                }
034                out.println("     Location where log statement called");
035                StackTraceElement[] stack = Thread.currentThread().getStackTrace();
036                if (stack != null && stack.length > 0) {
037                        for (int i = 0; i < stack.length; i++) {
038                                out.println("        " + stack[i].toString());
039                        }
040                }
041                if (apacheLogger != null) {
042                        apacheLog(level, writer.toString(), e);
043                } else if (slf4jLogger != null) {
044                        slf4jLog(level, writer.toString(), e);
045                }
046
047        }
048        public void logException(int level, String string) {
049                logException(level, string, (Exception) null);
050        }
051        public void logException(int level, String string, Exception e) {
052                List<String> list = new ArrayList<String>();
053                StringReader reader = new StringReader(string);
054                LineNumberReader in = new LineNumberReader(reader);
055                try {
056                while (true) {
057                        String line = in.readLine();
058                        if (line == null) { break; }
059                        list.add(line.trim());
060                        }
061                } catch (IOException e2) {
062                        list.add("Error generating log entry");
063                }
064                
065                logException(level, list, e);
066        }
067        protected void apacheLog(int level, String message, Exception e) {
068                switch (level) {
069                case FATAL :
070                        if (e == null) {
071                                apacheLogger.fatal(message);
072                        } else {
073                                apacheLogger.fatal(message, e);
074                        }
075                        break;
076                case ERROR :
077                        if (e == null) {
078                                apacheLogger.error(message);
079                        } else {
080                                apacheLogger.error(message, e);
081                        }
082                        break;                  
083                case WARN :
084                        if (e == null) {
085                                apacheLogger.warn(message);
086                        } else {
087                                apacheLogger.warn(message, e);
088                        }
089                        break;                  
090
091                case INFO :
092                        if (e == null) {
093                                apacheLogger.info(message);
094                        } else {
095                                apacheLogger.info(message, e);
096                        }
097                        break;
098                case DEBUG :
099                        if (e == null) {
100                                apacheLogger.debug(message);
101                        } else {
102                                apacheLogger.debug(message, e);
103                        }
104                        break;                  
105                case TRACE :
106                        if (e == null) {
107                                apacheLogger.trace(message);
108                        } else {
109                                apacheLogger.trace(message, e);
110                        }
111                        break;  
112                }
113        }
114        protected void slf4jLog(int level, String message, Exception e) {
115                switch (level) {
116                case FATAL :
117                        if (e == null) {
118                                slf4jLogger.error(message);
119                        } else {
120                                slf4jLogger.error(message, e);
121                        }
122                        break;
123                case ERROR :
124                        if (e == null) {
125                                slf4jLogger.error(message);

127                                slf4jLogger.error(message, e);

129                        break;                  
130                case WARN :
131                        if (e == null) {
132                                slf4jLogger.warn(message);
133                        } else {
134                                slf4jLogger.warn(message, e);
135                        }
136                        break;                  
137
138                case INFO :
139                        if (e == null) {
140                                slf4jLogger.info(message);
141                        } else {
142                                slf4jLogger.info(message, e);
143                        }
144                        break;
145                case DEBUG :
146                        if (e == null) {
147                                slf4jLogger.debug(message);
148                        } else {
149                                slf4jLogger.debug(message, e);
150                        }
151                        break;                  
152                case TRACE :
153                        if (e == null) {
154                                slf4jLogger.trace(message);
155                        } else {
156                                slf4jLogger.trace(message, e);
157                        }
158                        break;  
159                }
160        }
161        protected class Tester implements Runnable {
162                public void run() {
163                        try {
164                                level1();
165                        } catch (Exception e) {
166                                logException(INFO, "Trapping in method run", e);
167                        }
168                }
169                protected void level1() throws IOException {
170                        try {
171                                level2();
172                        } catch (Exception e) {
173                                logException(INFO, "Trapping in method level1", e);
174                                throw e;
175                        }
176                }
177                protected void level2() throws IOException {
178                        try {
179                                level3();
180                        } catch (Exception e) {
181                                logException(INFO, "Trapping in level2", e);
182                                throw e;
183                        }
184                }
185                protected void level3() throws IOException {
186                        IOException  e = new IOException("Triggering IOException");
187                        throw e;
188                }
189        }
