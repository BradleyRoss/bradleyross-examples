046        public class DummyExceptionProcessor implements ExceptionProcessor {
047                public List<String> getInformation(Exception e) {
048                        List<String> notes = new ArrayList<String>();
049                        if (e == null) {
050                                return notes;
051                        }
052                        if (SQLException.class.isAssignableFrom(e.getClass())) {
053                                SQLException e2 = (SQLException) e;
054                                notes.add("Exception is subclass of SQLException");
055                                notes.add("Vendor specific error code is " + Integer.toString(e2.getErrorCode()));
056                                String sqlState = e2.getSQLState();
057                                if (sqlState != null && sqlState.trim().length() > 0) {
058                                        notes.add("SQL State is " + sqlState);
059                                }
060                        } else if (SOAPFaultException.class.isAssignableFrom(e.getClass())) {
061                                SOAPFaultException e2 = (SOAPFaultException) e;
062                                notes.add("Exception is subclass of SOAPFaultException");
063                                SOAPFault fault = e2.getFault();
064                                if (fault == null) {
065                                        notes.add("SOAPFault object is null");
066                                } else {
067                                        notes.add("SOAPFault object is not null");
068                                }
069                        }
070                        return notes;
071                }
072        }

099        protected void logException(int level, List<String> value, Exception e) {
100                StringWriter writer = new StringWriter();
101                PrintWriter out = new PrintWriter(writer);
102                List<String> strings = new ArrayList<String>(value);
103                if (e != null && SQLException.class.isAssignableFrom(e.getClass())) {
104                        strings.add("Subclass of SQLException");
105                }
106                strings.addAll(extra.getInformation(e));
107                Iterator<String> iter = strings.iterator();
108                while (iter.hasNext()) {
109                        out.println(iter.next());
110                }
111                out.println("     Location where log statement called");
112                StackTraceElement[] stack = Thread.currentThread().getStackTrace();
113                if (stack != null && stack.length > 0) {
114                        for (int i = 0; i < stack.length; i++) {
115                                out.println("        " + stack[i].toString());
116                        }
117                }
118                if (apacheLogger != null) {
119                        apacheLog(level, writer.toString(), e);
120                } else if (slf4jLogger != null) {
121                        slf4jLog(level, writer.toString(), e);
122                }
123
124        }

126        public void logException(int level, String string, Exception e) {
127                List<String> list = new ArrayList<String>();
128                StringReader reader = new StringReader(string);
129                LineNumberReader in = new LineNumberReader(reader);
130                try {
131                        while (true) {
132                                String line = in.readLine();
133                                if (line == null) { break; }
134                                list.add(line.trim());
135                        }
136                } catch (IOException e2) {
137                        list.add("Error generating log entry");
138                }
139
140                logException(level, list, e);
141        }
142        protected void apacheLog(int level, String message, Exception e) {
143                switch (level) {
144                case FATAL :
145                        if (e == null) {
146                                apacheLogger.fatal(message);
147                        } else {
148                                apacheLogger.fatal(message, e);
149                        }
150                        break;
151                case ERROR :
152                        if (e == null) {
153                                apacheLogger.error(message);
154                        } else {
155                                apacheLogger.error(message, e);
156                        }
157                        break;                  
158                case WARN :
159                        if (e == null) {
160                                apacheLogger.warn(message);
161                        } else {
162                                apacheLogger.warn(message, e);
163                        }
164                        break;                  
165
166                case INFO :
167                        if (e == null) {
168                                apacheLogger.info(message);
169                        } else {
170                                apacheLogger.info(message, e);
171                        }
172                        break;
173                case DEBUG :
174                        if (e == null) {
175                                apacheLogger.debug(message);
176                        } else {
177                                apacheLogger.debug(message, e);
178                        }
179                        break;                  
180                case TRACE :
181                        if (e == null) {
182                                apacheLogger.trace(message);
183                        } else {
184                                apacheLogger.trace(message, e);
185                        }
186                        break;  
187                }
188        }

535        /**
536         * Demonstrates many of the features of the
537         * {@link ExceptionHelper} class.
538         * 
539         * @author Bradley Ross
540         *
541         */
542        protected class Tester implements Runnable {
543                protected ExceptionHelper helper = null;
544                public Tester(ExceptionHelper value) {
545                        helper = value;
546                }
547                public void run() {
548                        try {
549                                level1();
550                        } catch (Exception e) {
551                                helper.info("Trapping in method run", e);
552                        }
553                }
554                protected void level1() throws IOException {
555                        try {
556                                level2();
557                        } catch (Exception e) {
558                                helper.info("Trapping in method level1", e);
559                                throw e;
560                        }
561                }
562                protected void level2() throws IOException {
563                        try {
564                                level3();
565                        } catch (Exception e) {
566                                helper.info("Trap and throw in level 2 - level info", e);
567                                helper.warn("Trap and throw in level 2 - level warn", e);
568                                throw e;
569                        }
570                }
571                protected void level3() throws IOException {
572                        IOException  e = new IOException("Triggering IOException");
573                        throw e;
574                }
575        }
